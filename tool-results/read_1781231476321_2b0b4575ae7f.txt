   354→function PromptCard({ example }: { example: { title: string; objective: string; prompt: string } }) {
   355→  const [expanded, setExpanded] = useState(false)
   356→  const [copied, setCopied] = useState(false)
   357→
   358→  const handleCopy = async () => {
   359→    try {
   360→      await navigator.clipboard.writeText(example.prompt)
   361→      setCopied(true)
   362→      setTimeout(() => setCopied(false), 2500)
   363→    } catch {
   364→      const ta = document.createElement('textarea')
   365→      ta.value = example.prompt
   366→      document.body.appendChild(ta)
   367→      ta.select()
   368→      document.execCommand('copy')
   369→      document.body.removeChild(ta)
   370→      setCopied(true)
   371→      setTimeout(() => setCopied(false), 2500)
   372→    }
   373→  }
   374→
   375→  return (
   376→    <Card className="bg-card-bg/50 border-white/6 hover:border-lime/15 transition-all duration-300">
   377→      <CardContent className="p-5">
   378→        <div className="flex items-start justify-between gap-3 mb-3">
   379→          <div>
   380→            <h4 className="font-semibold text-foreground mb-1">{example.title}</h4>
   381→            <p className="text-sm text-muted-lavender">{example.objective}</p>
   382→          </div>
   383→          <Badge variant="outline" className="border-lime/20 text-lime text-xs flex-shrink-0">
   384→            Prompt
   385→          </Badge>
   386→        </div>
   387→
   388→        <button
   389→          onClick={() => setExpanded(!expanded)}
   390→          className="text-sm text-lime hover:text-lime-dark transition-colors flex items-center gap-1 mb-3"
   391→        >
   392→          {expanded ? 'Ocultar prompt' : 'Ver prompt completo'}
   393→          <ChevronDown
   394→            className={`size-4 transition-transform ${expanded ? 'rotate-180' : ''}`}
   395→          />
   396→        </button>
   397→
   398→        <AnimatePresence>
   399→          {expanded && (
   400→            <motion.div
   401→              initial={{ height: 0, opacity: 0 }}
   402→              animate={{ height: 'auto', opacity: 1 }}
   403→              exit={{ height: 0, opacity: 0 }}
   404→              transition={{ duration: 0.3 }}
   405→              className="overflow-hidden"
   406→            >
   407→              <div className="code-block">
   408→                <div className="flex items-center justify-between px-3 py-2 border-b border-white/6 bg-white/[0.02]">
   409→                  <span className="text-xs text-muted-lavender font-mono">prompt.txt</span>
   410→                  <button
   411→                    onClick={handleCopy}
   412→                    className={`text-xs flex items-center gap-1 px-2 py-1 rounded transition-colors ${
   413→                      copied
   414→                        ? 'text-lime bg-lime/10'
   415→                        : 'text-muted-lavender hover:text-foreground hover:bg-white/5'
   416→                    }`}
   417→                  >
   418→                    {copied ? <Check className="size-3" /> : <Copy className="size-3" />}
   419→                    {copied ? 'Copiado!' : 'Copiar'}
   420→                  </button>
   421→                </div>
   422→                <pre className="p-3 text-xs leading-relaxed font-mono text-foreground/80 whitespace-pre-wrap max-h-64 overflow-y-auto">
   423→                  {example.prompt}
   424→                </pre>
   425→              </div>
   426→            </motion.div>
   427→          )}
   428→        </AnimatePresence>
   429→      </CardContent>
   430→    </Card>
   431→  )
   432→}

   689→/* ─── Scroll to Top ─── */
   690→function ScrollToTop() {
   691→  const [visible, setVisible] = useState(false)
   692→
   693→  useEffect(() => {
   694→    const handleScroll = () => {
   695→      setVisible(window.scrollY > 600)
   696→    }
   697→    window.addEventListener('scroll', handleScroll)
   698→    return () => window.removeEventListener('scroll', handleScroll)
   699→  }, [])
   700→
   701→  return (
   702→    <AnimatePresence>
   703→      {visible && (
   704→        <motion.button
   705→          initial={{ opacity: 0, scale: 0.8 }}
   706→          animate={{ opacity: 1, scale: 1 }}
   707→          exit={{ opacity: 0, scale: 0.8 }}
   708→          onClick={() => window.scrollTo({ top: 0, behavior: 'smooth' })}
   709→          className="fixed bottom-6 right-6 z-50 w-10 h-10 rounded-full bg-lime text-navy flex items-center justify-center shadow-lg hover:bg-lime-dark transition-colors glow-lime"
   710→          aria-label="Voltar ao topo"
   711→        >
   712→          <ArrowUp className="size-5" />
   713→        </motion.button>
   714→      )}
   715→    </AnimatePresence>
   716→  )
   717→}
