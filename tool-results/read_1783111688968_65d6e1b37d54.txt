   153→function loadState(): WizardState {
   154→  if (typeof window === 'undefined') return { ...DEFAULT_STATE }
   155→  try {
   156→    const raw = localStorage.getItem(STORAGE_KEY)
   157→    if (raw) {
   158→      const parsed = JSON.parse(raw) as WizardState
   159→      // Ensure toggles has all keys (forward compat)
   160→      return {
   161→        ...DEFAULT_STATE,
   162→        ...parsed,
   163→        toggles: { ...DEFAULT_TOGGLES, ...parsed.toggles },
   164→      }
   165→    }
   166→  } catch {
   167→    // ignore
   168→  }
   169→  return { ...DEFAULT_STATE }
   170→}
   171→
   172→function saveState(state: WizardState) {
   173→  if (typeof window === 'undefined') return
   174→  try {
   175→    localStorage.setItem(STORAGE_KEY, JSON.stringify(state))
   176→  } catch {
   177→    // ignore
   178→  }
   179→}

   187→function generatePrompt(state: WizardState): string {
   188→  const preset = PRESETS.find((p) => p.id === state.selectedPreset)
   189→  const { fields, toggles, description } = state
   190→
   191→  // 1. Tipo
   192→  const tipoMap: Record<string, string> = {
   193→    tabela: 'Visualizador de Tabela Grande',
   194→    formulario: 'Formulário Padronizador',
   195→    calculadora: 'Calculadora / Ferramenta de Lógica',
   196→    consulta: 'Consulta de Situação / Painel de Indicadores',
   197→    cadastro: 'Sistema de Cadastro / Tramitação',
   198→    outro: 'Ferramenta Web Personalizada',
   199→  }
   200→  const tipo = tipoMap[state.selectedPreset] || 'Ferramenta Web'
   201→
   202→  // 2. Objetivo
   203→  const objetivo = description.trim() || preset?.description || '[Descreva o objetivo da ferramenta]'
   204→
   205→  // 3. Campos
   206→  let camposSection = ''
   207→  if (fields.length > 0) {
   208→    camposSection = fields.map((f) => {
   209→      const tipoLabel = FIELD_TYPE_LABELS[f.type]
   210→      let line = `- **${f.name}** (${tipoLabel})`
   211→      if (f.type === 'dropdown' && f.options.trim()) {
   212→        const opts = f.options.split(',').map((o) => o.trim()).filter(Boolean)
   213→        line += ` — Obrigatório, só aceita valores da lista: [${opts.join(', ')}]`
   214→      } else if (f.type === 'texto') {
   215→        line += ' — Texto livre, obrigatório'
   216→      } else if (f.type === 'data') {
   217→        line += ' — Formato DD/MM/AAAA, obrigatório'
   218→      } else if (f.type === 'numero') {
   219→        line += ' — Apenas dígitos, obrigatório'
   220→      }
   221→      return line
   222→    }).join('\n')
   223→  }
   224→
   225→  // 4. Funcionalidades
   226→  const funcionalidades: string[] = []
   227→  if (toggles.validacoes) funcionalidades.push('Validação de todos os campos obrigatórios antes de enviar')
   228→  if (toggles.designMobile) funcionalidades.push('Design responsivo (mobile-first)')
   229→  if (toggles.codigoUnico) funcionalidades.push('Todo o código em um único arquivo HTML')
   230→
   231→  // Table-specific features
   232→  if (state.selectedPreset === 'tabela') {
   233→    if (toggles.buscaInstantanea) funcionalidades.push('Busca instantânea que filtra em tempo real por múltiplas colunas')
   234→    if (toggles.filtrosColuna) funcionalidades.push('Filtros combinados por dropdown em colunas selecionadas')
   235→    if (toggles.paginacao) funcionalidades.push('Paginação com seletor de itens por página (10/25/50/100)')
   236→    if (toggles.ordenacao) funcionalidades.push('Ordenação clicável por coluna (ascendente/descendente)')
   237→    if (toggles.cabecalhoFixo) funcionalidades.push('Cabeçalho fixo ao rolar (sticky header)')
   238→  }
   239→
   240→  // 5. Integração com Google Sheets
   241→  let integracaoSection = ''
   242→  if (toggles.googleSheets) {
   243→    if (toggles.modoGoogle) {
   244→      integracaoSection = `Integre com Google Sheets usando HTML Service:
   245→- Crie dois arquivos: Code.gs e Index.html
   246→- No Code.gs, implemente doGet() que retorna HtmlService.createHtmlOutputFromFile('Index')
   247→- No Code.gs, implemente funções com google.script.run para ler e gravar dados na planilha
   248→- No Index.html, coloque todo o código HTML/CSS/JS da interface
   249→- Use SpreadsheetApp.openById() para acessar a planilha por ID
   250→- Dados de exemplo FICTÍCIOS embutidos no código para demonstração`
   251→    } else {
   252→      integracaoSection = `Integre com Google Sheets usando Apps Script:
   253→- No arquivo HTML, implemente funções para enviar dados via fetch() para a URL do Apps Script (web app)
   254→- No Code.gs, implemente doGet() para ler dados da planilha e retornar como JSON
   255→- No Code.gs, implemente doPost() para receber dados do formulário e gravar na planilha
   256→- Use ContentService.createTextOutput() com JSON.stringify() para retornar dados
   257→- Use SpreadsheetApp.getActiveSpreadsheet() para acessar a planilha
   258→- Dados de exemplo FICTÍCIOS embutidos no código para demonstração`
   259→    }
   260→  }
   261→
   262→  // 6. Preset-specific instructions
   263→  let presetSpecific = ''
   264→  if (state.selectedPreset === 'tabela') {
   265→    presetSpecific = `\n## Dados de Exemplo
   266→Inclua pelo menos 20 registros fictícios no estilo de dados de servidor público universitário (ex: nome, matrícula, setor, cargo, situação, data de admissão). Os dados devem ser realistas mas completamente fictícios.`
   267→  } else if (state.selectedPreset === 'formulario') {
   268→    presetSpecific = `\n## Regras do Formulário Padronizador
   269→- Todos os dropdowns são obrigatórios e só aceitam valores da lista predefinida
   270→- Campos derivados devem ser preenchidos automaticamente com base na seleção de outros campos
   271→- Validação completa antes de enviar — destaque visualmente os campos com erro
   272→- Feedback visual claro: campos preenchidos corretamente ficam verdes, com erro ficam vermelhos
   273→- Botão "Limpar" para resetar todo o formulário`
   274→  } else if (state.selectedPreset === 'calculadora') {
   275→    presetSpecific = `\n## Lógica da Calculadora
   276→- Todos os cálculos devem ser atualizados em tempo real conforme o usuário altera os campos
   277→- Mostre o resultado de forma destacada e fácil de ler
   278→- Inclua validação para impedir valores inválidos (ex: divisão por zero, campos vazios)`
   279→  }
   280→
   281→  // Assemble
   282→  let prompt = `Crie uma ${tipo} com as seguintes especificações:
   283→
   284→## Objetivo
   285→${objetivo}`
   286→
   287→  if (camposSection) {
   288→    prompt += `\n\n## Campos\n${camposSection}`
   289→  }
   290→
   291→  if (funcionalidades.length > 0) {
   292→    prompt += `\n\n## Funcionalidades\n${funcionalidades.map((f) => `- ${f}`).join('\n')}`
   293→  }
   294→
   295→  if (integracaoSection) {
   296→    prompt += `\n\n## Integração com Google Sheets\n${integracaoSection}`
   297→  }
   298→
   299→  if (presetSpecific) {
   300→    prompt += `\n${presetSpecific}`
   301→  }
   302→
   303→  prompt += `\n\n## Requisitos
   304→- Todo o código em ${toggles.modoGoogle && toggles.googleSheets ? 'dois arquivos (Code.gs e Index.html)' : 'um único arquivo HTML'}
   305→- Dados de exemplo FICTÍCIOS (nunca dados reais)
   306→- Comentários em português em todo o código
   307→- Design responsivo (mobile-first)
   308→- Visual moderno com tema escuro e cores institucionais
   309→- Funcional imediatamente — não deixe placeholders ou TODOs`
   310→
   311→  return prompt
   312→}

   459→function StepCampos({
   460→  state,
   461→  setState,
   462→}: {
   463→  state: WizardState
   464→  setState: (s: WizardState) => void
   465→}) {
   466→  const addField = () => {
   467→    setState({
   468→      ...state,
   469→      fields: [...state.fields, { id: newFieldId(), name: '', type: 'texto', options: '' }],
   470→    })
   471→  }
   472→
   473→  const updateField = (id: string, updates: Partial<FieldDef>) => {
   474→    setState({
   475→      ...state,
   476→      fields: state.fields.map((f) => (f.id === id ? { ...f, ...updates } : f)),
   477→    })
   478→  }
   479→
   480→  const removeField = (id: string) => {
   481→    setState({
   482→      ...state,
   483→      fields: state.fields.filter((f) => f.id !== id),
   484→    })
   485→  }
   486→
   487→  // Suggested fields based on preset
   488→  const presetSuggestions: Record<string, string[]> = {
   489→    tabela: ['Nome do Servidor', 'Matrícula', 'Setor', 'Cargo', 'Situação', 'Data de Admissão'],
   490→    formulario: ['Nome do Solicitante', 'Setor de Origem', 'Tipo de Pedido', 'Descrição', 'Data do Pedido'],
   491→    calculadora: ['Valor Base', 'Percentual', 'Resultado'],
   492→    consulta: ['Protocolo', 'Nome', 'Situação', 'Data'],
   493→    cadastro: ['Nome Completo', 'Documento', 'Setor', 'Assunto', 'Data de Entrada', 'Status'],
   494→    outro: [],
   495→  }
   496→
   497→  const suggestions = presetSuggestions[state.selectedPreset] || []
   498→  const unusedSuggestions = suggestions.filter(
   499→    (s) => !state.fields.some((f) => f.name.toLowerCase() === s.toLowerCase())
   500→  )
   501→
   502→  return (
   503→    <div className="space-y-6">
   504→      <div>
   505→        <h3 className="text-lg font-semibold text-foreground mb-1">
   506→          Campos e dados
   507→        </h3>
   508→        <p className="text-sm text-muted-lavender">
   509→          Defina os campos que sua ferramenta precisa ter
   510→        </p>
   511→      </div>
   512→
   513→      {/* Quick-add suggestions */}
   514→      {unusedSuggestions.length > 0 && (
   515→        <div>
   516→          <label className="block text-xs font-medium text-muted-lavender mb-2">
   517→            Sugestões rápidas (clique para adicionar)
   518→          </label>
   519→          <div className="flex flex-wrap gap-2">
   520→            {unusedSuggestions.map((s) => (
   521→              <button

   845→  state,
   846→  onReset,
   847→}: {
   848→  state: WizardState
   849→  onReset: () => void
   850→}) {
   851→  const [copied, setCopied] = useState(false)
   852→  const prompt = generatePrompt(state)
   853→
   854→  const handleCopy = async () => {
   855→    try {
   856→      await navigator.clipboard.writeText(prompt)
   857→    } catch {
   858→      const ta = document.createElement('textarea')
   859→      ta.value = prompt
   860→      document.body.appendChild(ta)
   861→      ta.select()
   862→      document.execCommand('copy')
   863→      document.body.removeChild(ta)
   864→    }
   865→    setCopied(true)
   866→    toast.success('Prompt copiado para a área de transferência!')
   867→    setTimeout(() => setCopied(false), 3000)
   868→  }
   869→
   870→  const handleDownload = () => {
   871→    const blob = new Blob([prompt], { type: 'text/plain;charset=utf-8' })
   872→    const url = URL.createObjectURL(blob)
   873→    const a = document.createElement('a')
   874→    a.href = url
   875→    const tipo = state.selectedPreset || 'ferramenta'
   876→    a.download = `prompt-${tipo}-${Date.now()}.txt`
   877→    document.body.appendChild(a)
   878→    a.click()
   879→    document.body.removeChild(a)
   880→    URL.revokeObjectURL(url)
   881→    toast.success('Arquivo .txt baixado com sucesso!')

   891→
