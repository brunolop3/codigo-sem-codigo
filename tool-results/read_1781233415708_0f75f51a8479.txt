    65→    65→      const sectionIds = links.map((l) => l.href.replace('#', ''))

    71→    71→            setActiveSection(sectionIds[i])

    77→    77→    }

    80→    80→  }, [])
    81→    81→
    82→    82→  return (
    83→    83→    <nav className="sticky top-0 z-50 w-full border-b border-white/6 bg-background/80 backdrop-blur-xl">
    84→    84→      <div className="max-w-6xl mx-auto px-4 sm:px-6 lg:px-8">
    85→    85→        <div className="flex items-center justify-between h-14">
    86→    86→          <a href="#" className="flex items-center gap-2 font-bold text-lg">

    91→    91→              Código<span className="text-lime">semCódigo</span>
    92→    92→            </span>
    93→    93→          </a>
    94→    94→          <div className="hidden md:flex items-center gap-1">
    95→    95→            {links.map((link) => {
    96→    96→              const id = link.href.replace('#', '')
    97→    97→              const isActive = activeSection === id
    98→    98→              return (
    99→    99→                <a
   100→   100→                  key={link.href}
   101→   101→                  href={link.href}
   102→   102→                  className={`px-3 py-1.5 text-sm rounded-md transition-all duration-200 ${
   103→   103→                    isActive ? 'text-lime bg-lime/10 font-medium' : 'text-muted-lavender hover:text-foreground hover:bg-white/5'
   104→   104→                  }`}
   105→   105→                >
   106→   106→                  {link.label}
   107→   107→                </a>
   108→   108→              )
   109→   109→            })}
   110→   110→          </div>
   111→   111→          <button className="md:hidden p-2 text-muted-lavender hover:text-foreground" onClick={() => setOpen(!open)} aria-label="Menu">
   112→   112→            <svg width="20" height="20" viewBox="0 0 20 20" fill="none">
   113→   113→              <path d={open ? 'M5 5L15 15M15 5L5 15' : 'M3 6H17M3 10H17M3 14H17'} stroke="currentColor" strokeWidth="1.5" strokeLinecap="round" />
   114→   114→            </svg>
   115→   115→          </button>
   116→   116→        </div>
   117→   117→        <AnimatePresence>
   118→   118→          {open && (
   119→   119→            <motion.div initial={{ height: 0, opacity: 0 }} animate={{ height: 'auto', opacity: 1 }} exit={{ height: 0, opacity: 0 }} className="md:hidden overflow-hidden border-t border-white/6">
   120→   120→              <div className="py-3 space-y-1">
   121→   121→                {links.map((link) => {
   122→   122→                  const id = link.href.replace('#', '')
   123→   123→                  const isActive = activeSection === id
   124→   124→                  return (
   125→   125→                    <a key={link.href} href={link.href} onClick={() => setOpen(false)} className={`block px-3 py-2 text-sm rounded-md transition-all ${isActive ? 'text-lime bg-lime/10 font-medium' : 'text-muted-lavender hover:text-foreground hover:bg-white/5'}`}>
   126→   126→                      {link.label}
   127→   127→                    </a>
   128→   128→                  )
   129→   129→                })}
   130→   130→              </div>
   131→   131→            </motion.div>
   132→   132→          )}
   133→   133→        </AnimatePresence>
   134→   134→      </div>
   135→   135→    </nav>
   136→   136→  )
   137→   137→}

   796→   796→function ScrollToTop() {
   797→   797→  const [visible, setVisible] = useState(false)
   798→   798→  useEffect(() => {
   799→   799→    const handleScroll = () => setVisible(window.scrollY > 600)
   800→   800→    window.addEventListener('scroll', handleScroll)
   801→   801→    return () => window.removeEventListener('scroll', handleScroll)
   802→   802→  }, [])
   803→   803→  return (
   804→   804→    <AnimatePresence>
   805→   805→      {visible && (
   806→   806→        <motion.button
   807→   807→          initial={{ opacity: 0, scale: 0.8 }}
   808→   808→          animate={{ opacity: 1, scale: 1 }}
   809→   809→          exit={{ opacity: 0, scale: 0.8 }}
   810→   810→          onClick={() => window.scrollTo({ top: 0, behavior: 'smooth' })}
   811→   811→          className="fixed bottom-6 right-6 z-50 w-10 h-10 rounded-full bg-lime text-navy flex items-center justify-center shadow-lg hover:bg-lime-dark transition-colors glow-lime"
   812→   812→          aria-label="Voltar ao topo"
   813→   813→        >
   814→   814→          <ArrowUp className="size-5" />
   815→   815→        </motion.button>
   816→   816→      )}
   817→   817→    </AnimatePresence>

   819→   819→}
