    58→/* ─── Navigation ─── */
    59→function Navigation() {
    60→  const [open, setOpen] = useState(false)
    61→  const [activeSection, setActiveSection] = useState<string | null>(null)
    62→  const links = [
    63→    { href: '#guide', label: 'Guia' },
    64→    { href: '#demo', label: 'Demo' },
    65→    { href: '#dicionario', label: 'Dicionário' },
    66→    { href: '#enade', label: 'Exemplo' },
    67→    { href: '#levels', label: 'Níveis' },
    68→    { href: '#prompt', label: 'Prompt' },
    69→    { href: '#builder', label: 'Construtor' },
    70→    { href: '#ideas', label: 'Ideias' },
    71→    { href: '#bastidores', label: 'Bastidores' },
    72→    { href: '#seguranca', label: 'Segurança' },
    73→    { href: '#tips', label: 'Dicas' },
    74→  ]
    75→
    76→  useEffect(() => {
    77→    const handleScroll = () => {
    78→      const sectionIds = links.map((l) => l.href.replace('#', ''))
    79→      for (let i = sectionIds.length - 1; i >= 0; i--) {
    80→        const el = document.getElementById(sectionIds[i])
    81→        if (el) {
    82→          const rect = el.getBoundingClientRect()
    83→          if (rect.top <= 120) {
    84→            setActiveSection(sectionIds[i])
    85→            return
    86→          }
    87→        }
    88→      }
    89→      setActiveSection(null)
    90→    }
    91→    window.addEventListener('scroll', handleScroll, { passive: true })
    92→    return () => window.removeEventListener('scroll', handleScroll)
    93→  }, [])
    94→
    95→  return (
    96→    <nav className="sticky top-0 z-50 w-full border-b border-white/6 bg-background/80 backdrop-blur-xl">
    97→      <div className="max-w-6xl mx-auto px-4 sm:px-6 lg:px-8">
    98→        <div className="flex items-center justify-between h-14">
    99→          <a href="#" className="flex items-center gap-2 font-bold text-lg">
   100→            <div className="w-7 h-7 rounded-lg bg-lime flex items-center justify-center">
   101→              <Zap className="size-4 text-navy" />
   102→            </div>
   103→            <span className="text-foreground">
   104→              Código<span className="text-lime">semCódigo</span>
   105→            </span>
   106→          </a>
   107→          <div className="hidden lg:flex items-center gap-1">
   108→            {links.map((link) => {
   109→              const id = link.href.replace('#', '')
   110→              const isActive = activeSection === id
   111→              return (
   112→                <a
   113→                  key={link.href}
   114→                  href={link.href}
   115→                  className={`px-2.5 py-1.5 text-xs rounded-md transition-all duration-200 ${
   116→                    isActive ? 'text-lime bg-lime/10 font-medium' : 'text-muted-lavender hover:text-foreground hover:bg-white/5'
   117→                  }`}
   118→                >
   119→                  {link.label}
   120→                </a>
   121→              )
   122→            })}
   123→          </div>
   124→          <button className="lg:hidden p-2 text-muted-lavender hover:text-foreground" onClick={() => setOpen(!open)} aria-label="Menu">
   125→            <svg width="20" height="20" viewBox="0 0 20 20" fill="none">
   126→              <path d={open ? 'M5 5L15 15M15 5L5 15' : 'M3 6H17M3 10H17M3 14H17'} stroke="currentColor" strokeWidth="1.5" strokeLinecap="round" />
   127→            </svg>
   128→          </button>
   129→        </div>
   130→        <AnimatePresence>
   131→          {open && (
   132→            <motion.div initial={{ height: 0, opacity: 0 }} animate={{ height: 'auto', opacity: 1 }} exit={{ height: 0, opacity: 0 }} className="lg:hidden overflow-hidden border-t border-white/6">
   133→              <div className="py-3 space-y-1">
   134→                {links.map((link) => {
   135→                  const id = link.href.replace('#', '')
   136→                  const isActive = activeSection === id
   137→                  return (
   138→                    <a key={link.href} href={link.href} onClick={() => setOpen(false)} className={`block px-3 py-2 text-sm rounded-md transition-all ${isActive ? 'text-lime bg-lime/10 font-medium' : 'text-muted-lavender hover:text-foreground hover:bg-white/5'}`}>
   139→                      {link.label}
   140→                    </a>
   141→                  )
   142→                })}
   143→              </div>
   144→            </motion.div>
   145→          )}
   146→        </AnimatePresence>
   147→      </div>
   148→    </nav>
   149→  )
   150→}

   440→function PromptCard({ example }: { example: { title: string; objective: string; prompt: string; step: number; stepLabel: string } }) {
   441→  const [expanded, setExpanded] = useState(false)
   442→  const [copied, setCopied] = useState(false)
   443→
   444→  const handleCopy = async () => {

   449→      ta.value = example.prompt

  1040→function ScrollToTop() {

  1042→  useEffect(() => {
  1043→    const handleScroll = () => setVisible(window.scrollY > 600)

  1045→    return () => window.removeEventListener('scroll', handleScroll)
  1046→  }, [])
  1047→  return (
  1048→    <AnimatePresence>
  1049→      {visible && (
  1050→        <motion.button
  1051→          initial={{ opacity: 0, scale: 0.8 }}
  1052→          animate={{ opacity: 1, scale: 1 }}
  1053→          exit={{ opacity: 0, scale: 0.8 }}
  1054→          onClick={() => window.scrollTo({ top: 0, behavior: 'smooth' })}

  1057→        >
  1058→          <ArrowUp className="size-5" />
